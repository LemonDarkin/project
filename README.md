def anargamma(s1, s2):
   
    s1 = s1.replace(" ", "").lower()
    s2 = s2.replace(" ", "").lower()
    
    return sorted(s1) == sorted(s2)
    
a = input("Введите первую строку: ")
b = input("Введите вторую строку: ")

if anargamma(a, b):
    print("Строки являются анаграммами")
else:
    print("Строки не являются анаграммами")
